#include <bits/stdc++.h>

using namespace std;
using pii = pair<int, int>;

#define rep(i, a, b) for (auto i = (a); i <= decltype(i)(b); ++i)

inline auto lowb(auto x) { return x & -x; }
template <typename T>
struct Fenwick {
    int n;
    std::vector<T> c;

    Fenwick(auto n_ = 0) {
        init(n_);
    }
    void init(auto n_) {
        n = n_;
        c.assign(n, T{});
    }
    T ask(auto x) {
        T ret{};
        for (; x; x -= lowb(x)) ret += c[x];
        return ret;
    }
    void add(auto x, auto k) {
        for (; x <= n; x += lowb(x)) c[x] += k;
    }
};

signed main() {
	cin.tie(nullptr)->sync_with_stdio(false);

	int n, q; cin >> n >> q;
	vector<pii> v(n + 10);
	rep (i, 1, n) cin >> v[i].first >> v[i].second;

    unordered_map<int, int> mp;
	vector<int> pos(2 * n + 10);
    auto discrete = [&]() {
        rep (i, 1, n) { pos[i] = v[i].first; pos[n + i] = v[i].second; }
        sort(pos.begin() + 1, pos.begin() + 2 * n + 1);
        auto lst = unique(pos.begin() + 1, pos.begin() + 2 * n + 1) - pos.begin() - 1;
        rep (i, 1, lst) mp[pos[i]] = i;
        return lst;
    };
    int len = discrete();

    Fenwick<int> f(len + 10);
    vector<int> bg(len + 10);
    rep (i, 1, n) {
        f.add(mp[v[i].first], 1); f.add(mp[v[i].second] + 1, -1);
        ++bg[mp[v[i].first]];
    }

    while (q--) {
        int x; cin >> x;
        if (mp.contains(x))
            cout << f.ask(mp[x]) << endl;
        else {
            auto p = upper_bound(pos.begin() + 1, pos.begin() + len + 1, x);
            auto t = *p;
            if (p == pos.begin() + len + 1) cout << 0 << endl;
            else cout << f.ask(mp[t]) - bg[mp[t]] << endl;
        }
    }

	return 0;
}
